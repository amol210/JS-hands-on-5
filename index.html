2-Count the Heads

    var Count_Occ = (s) => {
        let arr = s.split("").sort();
        let heads = "";
        let visit = [];
        let k = 0;
        let res = "";
        for(let i=0;i < arr.length;i++)
        {
          let count = 0;
          for(let j=i;j < arr.length;j++)
          {
            if(arr[i] === arr[j])
            {
              heads = arr[i];
              count++;
            }
          }
          if(!visit.includes(heads) && count >= 2)
          {
            visit[k] = heads;
            visit[k+1] = count;
            k= k + 2;
          }
        }
          res = visit.join("");
        return res;
    };

    var Game_Winner = (S) => 
    {
      let Aletter=0,Dletter=0;
      for(let i=0;i < S.length;i++){
        if(S[i]==="A")
          Aletter+=1;
        else if(S[i]==="D")
        Dletter+=1;
      }
      if(Aletter < Dletter)
      return "Danish";
      else if(Aletter > Dletter)
      return "Aditya";
      else if(Aletter === Dletter)
      return "Draw";
    };
